cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_pac1934 LANGUAGES MikroC)
else()
    project(lib_pac1934 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_pac1934 STATIC
        src/pac1934.c
        include/pac1934.h
)
add_library(Click.Pac1934  ALIAS lib_pac1934)



target_include_directories(lib_pac1934 PUBLIC
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_pac1934 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_pac1934 PUBLIC MikroSDK.Driver)
